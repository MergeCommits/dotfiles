### Mapper

# Navigating the page
map gg LinkHints.activateHover
map gt goUp
map gT goToRoot

# Marks
map , Marks.activate
map < Marks.activateCreateMode
map > Marks.clearLocal

# Scrolling
map l scrollUp
map u scrollToTop
map U scrollToBottom

# Tabs
shortcut userCustomized1 command="restoreTab"
shortcut userCustomized2 command="removeTab"

# Tab Management
map J moveTabLeft
map L moveTabRight

# Mute
map m toggleMuteTab

map e Vomnibar.activateUrl

map H Vomnibar.activateHistoryInNewTab
